## Installation
<Tabs items={['pnpm', 'npm', 'yarn']} groupId="language">

    ```bash

    ```  
  </Tab>
  <Tab value="npm">

  </Tab>
  <Tab value="yarn">
    ```bash
    yarn add jum-react
    ```
  </Tab>

## Usage
```tsx
import { Jum } from 'jum-react'

<Jum>
  <img src="image.webp" alt="" />
</Jum>
```
