## Installation
<Tabs items={['pnpm', 'npm', 'yarn']} groupId="language">
  <Tab value="pnpm">
    ```bash
    pnpm add jum-core
    ```  
  </Tab>
  <Tab value="npm">
    ```bash
    npm i jum-core
    ```
  </Tab>
  <Tab value="yarn">
    ```bash
    yarn add jum-core
    ```
  </Tab>
</Tabs>

## Usage
```js
import { jum } from 'jum-core'

const jumer = jum(element, options)
```
